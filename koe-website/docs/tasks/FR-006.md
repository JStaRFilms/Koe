## 🏗️ Implementation Notes

### GitHub API
```typescript
// lib/github.ts
export async function getRepoStars(): Promise<number> {
  const response = await fetch(
    'https://api.github.com/repos/GIGAHAT1994/whisper_alt',
    { next: { revalidate: 3600 } }
  );
  
  const data = await response.json();
  return data.stargazers_count;
}

// Format number (1200 → 1.2k)
export function formatStarCount(count: number): string {
  if (count >= 1000) {
    return `${(count / 1000).toFixed(1)}k`;
  }
  return count.toString();
}
```

### Star Badge Component
```typescript
// components/github-star-badge.tsx
'use client';

import { Github, Star } from "lucide-react";
import { useGitHubStars } from "@/hooks/use-github-stars";
import { Button } from "@/components/ui/button";

export function GitHubStarBadge() {
  const { stars, loading, error } = useGitHubStars();

  if (loading || error || stars === null) {
    return (
      <Button variant="outline" size="sm" asChild>
        <a href="https://github.com/GIGAHAT1994/whisper_alt" target="_blank" rel="noopener noreferrer">
          <Github className="w-4 h-4 mr-2" />
          Star on GitHub
        </a>
      </Button>
    );
  }

  return (
    <Button variant="outline" size="sm" asChild>
      <a href="https://github.com/GIGAHAT1994/whisper_alt" target="_blank" rel="noopener noreferrer" className="gap-2">
        <Github className="w-4 h-4" />
        <span>Star</span>
        <span className="flex items-center gap-1 text-muted-foreground">
          <Star className="w-3 h-3 fill-current" />
          {formatStarCount(stars)}
        </span>
      </a>
    </Button>
  );
}
```
